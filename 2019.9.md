``` sql
  SELECT 1
  WHILE @@ROWCOUNT > 0
  BEGIN
    DELETE TOP 100
    FROM LargeTable
  END
  
  DoItAgain:
    DELETE TOP 100
    FROM ExampleTable
    IF @@ROWCOUNT > 0
  GOTO DoItAgain
```
